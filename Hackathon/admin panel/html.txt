        // Add new drug
        function addDrug(drugData) {
            const newId = inventoryData.length > 0 ? Math.max(...inventoryData.map(d => d.id)) + 1 : 1;
            const newDrug = {
                id: newId,
                ...drugData
            };
            
            inventoryData.push(newDrug);
            renderInventoryTable(inventoryData);
            closeModal();
            showAlert('Drug Added!', `${newDrug.name} ${newDrug.dosage} has been added to inventory.`, 'success');
        }
        
        // Update existing drug
        function updateDrug(drugId, drugData) {
            const index = inventoryData.findIndex(d => d.id === drugId);
            if (index !== -1) {
                inventoryData[index] = {
                    id: drugId,
                    ...drugData
                };
                
                renderInventoryTable(inventoryData);
                closeModal();
                showAlert('Inventory Updated!', `${drugData.name} ${drugData.dosage} has been updated.`, 'success');
            }
        }

        // Delete drug

            if (confirm('Are you sure you want to remove this drug from inventory?')) {
                const index = inventoryData.findIndex(d => d.id === drugId);
                if (index !== -1) {
                    const deletedDrug = inventoryData[index];
                    inventoryData.splice(index, 1);
                    
                    renderInventoryTable(inventoryData);
                    showAlert('Drug Removed!', `${deletedDrug.name} ${deletedDrug.dosage} has been removed from inventory.`, 'success');
                }
            }
        }

                                 order.status === 'processing' ? 'Processing' : 'Ready for Pickup';
                
                const statusClass = order.status === 'pending' ? 'status-pending' : 
                                  order.status === 'processing' ? 'status-processing' : 'status-ready';
                
                const statusSpan = document.createElement('span');
                statusSpan.className = statusClass;
                statusSpan.textContent = statusText;
                statusElement.appendChild(statusSpan);
                
                const itemsContainer = document.getElementById('orderDetailItems');
                itemsContainer.innerHTML = '';
                
                order.items.forEach(item => {
                    const itemElement = document.createElement('div');
                    itemElement.className = 'order-item';
                    itemElement.innerHTML = `
                        <div>
                            <div class="item-name">${item.name}</div>
                            <div class="item-detail">Qty: ${item.quantity}</div>
                        </div>
                        <div>₹${(item.price * item.quantity).toFixed(2)}</div>
                    `;
                    itemsContainer.appendChild(itemElement);
                });
                
                // Set prescription image if available
                if (order.prescriptionAvailable) {
                    prescriptionImage.src = 'https://placehold.co/600x400/2a9d8f/white?text=Prescription+Image';
                    prescriptionImage.style.display = 'block';
                    prescriptionPlaceholder.style.display = 'none';
                } else {
                    prescriptionImage.style.display = 'none';
                    prescriptionPlaceholder.style.display = 'block';
                }
                
                orderModal.classList.add('show');
            }
        }
        
        // Toggle prescription image
        function togglePrescription() {
            if (prescriptionImage.style.display === 'none') {
                prescriptionImage.style.display = 'block';
                prescriptionPlaceholder.style.display = 'none';
            } else {
                prescriptionImage.style.display = 'none';
                prescriptionPlaceholder.style.display = 'block';
            }
        }
        
        // Notify customer about missing prescription
        function notifyCustomer() {
            showAlert('Customer Notified!', 'The customer has been notified about the missing prescription.', 'success');
        }
        
        // Process order
        function processOrder() {
            showAlert('Order Processed!', 'The order has been successfully processed.', 'success');
            closeOrderModal();
        }
        
        // Show alert
        function showAlert(title, description, type) {
            alertTitle.textContent = title;
            alertDesc.textContent = description;
            
            // Set alert type
            alertPopup.className = 'alert-popup';
            if (type === 'success') {
                alertPopup.classList.add('show');
                alertIcon.className = 'fas fa-check';
            } else if (type === 'error') {
                alertPopup.classList.add('show', 'error');
                alertIcon.className = 'fas fa-times';
            }
            
            // Auto hide after 3 seconds
            setTimeout(() => {
                hideAlert();
            }, 3000);
        }
        
        // Hide alert
        function hideAlert() {
            alertPopup.classList.remove('show');
        }
        
        // Initialize the page
        init();
    </script>
